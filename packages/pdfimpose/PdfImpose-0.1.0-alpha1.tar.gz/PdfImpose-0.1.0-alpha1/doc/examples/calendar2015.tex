  \usepackage[
      pdftitle={2015 Calendar},

  ]{hyperref}

          if={(Sunday) [black!50]}
        }]
        \calendar [dates=\theyear-\month-01 to \theyear-\month-last];
      \end{tikzpicture}
    }

    \pagebreak

  }
